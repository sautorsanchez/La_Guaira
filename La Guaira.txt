<script>
    // Configuración de los Sectores Sismogénicos (Zonas de conflicto) basados en la imagen
    const SECTOR_DATA = [
        { id: "playa-verde", name: "Playa Verde", x: 300, y: 550, radius: 100, damageCount: 412, severity: "moderado", elev: "12 m", desc: "Zonas de viviendas de baja elevación costera. Licuación de suelos localizada." },
        { id: "brisas-aeropuerto", name: "Brisas del Aeropuerto", x: 180, y: 380, radius: 120, damageCount: 685, severity: "severo", elev: "65 m", desc: "Viviendas en laderas con alta pendiente. Deslizamientos arrastraron cimientos de soporte." },
        { id: "aeropuerto", name: "Maiquetía (Aeropuerto)", x: 450, y: 440, radius: 250, damageCount: 154, severity: "leve", elev: "45 m", desc: "Pistas y hangares. Fracturas longitudinales en asfalto de rampa norte y terminal nacional." },
        { id: "brisas-maiquetia", name: "Brisas de Maiquetía", x: 420, y: 320, radius: 110, damageCount: 890, severity: "severo", elev: "115 m", desc: "Zona crítica. Alta densidad estructural afectada por el empuje cinemático del terreno montañoso." },
        { id: "hospital", name: "Hospital Martín Vegas", x: 580, y: 220, radius: 90, damageCount: 123, severity: "moderado", elev: "95 m", desc: "Infraestructura médica. Colapso parcial de mampostería perimetral exterior." },
        { id: "los-olivos", name: "Los Olivos", x: 680, y: 150, radius: 130, damageCount: 574, severity: "severo", elev: "140 m", desc: "Sector de ladera alta. Gran porcentaje de asentamiento de terreno diferencial." },
        { id: "catia-la-mar", name: "Catia La Mar Centro", x: 820, y: 280, radius: 160, damageCount: 583, severity: "moderado", elev: "28 m", desc: "Zona urbana consolidada. Agrietamiento estructural severo en edificaciones multifamiliares de mediana altura." }
    ];

    // Simular Estructuras Dañadas (Miles de puntos basados en densidad de sectores)
    const GENERATED_BUILDINGS = [];
    
    function generateScenario() {
        GENERATED_BUILDINGS.length = 0;
        SECTOR_DATA.forEach(sector => {
            // Generar puntos alrededor del centro de cada sector siguiendo distribución gaussiana
            for (let i = 0; i < sector.damageCount; i++) {
                const angle = Math.random() * Math.PI * 2;
                const dist = Math.pow(Math.random(), 1.5) * sector.radius;
                
                const bx = sector.x + Math.cos(angle) * dist;
                const by = sector.y + Math.sin(angle) * dist;
                
                // Determinar nivel de gravedad del daño individual
                let severity = "moderate";
                const rand = Math.random();
                if (sector.severity === "severo") {
                    severity = rand > 0.3 ? "severe" : (rand > 0.1 ? "moderate" : "light");
                } else if (sector.severity === "moderado") {
                    severity = rand > 0.6 ? "severe" : (rand > 0.2 ? "moderate" : "light");
                } else {
                    severity = rand > 0.8 ? "severe" : (rand > 0.4 ? "moderate" : "light");
                }

                // Tipo de uso
                const uses = ["Residencial", "Comercial", "Educativo", "Gubernamental"];
                const usage = uses[Math.floor(Math.random() * uses.length)];

                // Generar descripción técnica simulada para didáctica
                let desc = "Fisuras diagonales en muros estructurales. Asentamiento leve.";
                if (severity === "severe") {
                    desc = "Pandeo de columnas, agrietamiento de losas de carga, falla total de la cimentación.";
                } else if (severity === "light") {
                    desc = "Grietas menores en acabados. Desprendimiento superficial de pintura y mampostería estética.";
                }

                GENERATED_BUILDINGS.push({
                    id: `B-${sector.id.slice(0, 3).toUpperCase()}-${1000 + i}`,
                    x: bx,
                    y: by,
                    severity: severity,
                    sectorName: sector.name,
                    elev: Math.floor(parseInt(sector.elev) + (Math.random() * 20 - 10)) + " m",
                    usage: usage,
                    desc: desc
                });
            }
        });
    }

    // Inicializar lista lateral
    function populateSectorList() {
        const listContainer = document.getElementById("sectors-list");
        listContainer.innerHTML = "";
        
        SECTOR_DATA.forEach(sector => {
            const badgeColor = sector.severity === "severo" ? "bg-red-500/10 text-red-400 border-red-500/20" : 
                               sector.severity === "moderado" ? "bg-orange-500/10 text-orange-400 border-orange-500/20" : 
                               "bg-yellow-500/10 text-yellow-400 border-yellow-500/20";
            
            const sectorElement = document.createElement("div");
            sectorElement.className = "group bg-slate-900/60 hover:bg-slate-800/80 p-3 rounded-xl border border-slate-800/80 hover:border-slate-700 cursor-pointer transition flex items-center justify-between";
            sectorElement.dataset.sectorId = sector.id;
            
            sectorElement.innerHTML = `
                <div class="space-y-0.5">
                    <div class="text-xs font-semibold text-slate-200 group-hover:text-white flex items-center gap-1.5">
                        <span class="w-1.5 h-1.5 rounded-full bg-slate-500 group-hover:bg-cyan-400 transition"></span>
                        ${sector.name}
                    </div>
                    <div class="text-[10px] text-slate-400 font-mono">Elevación prom: ${sector.elev}</div>
                </div>
                <div class="text-right space-y-1">
                    <span class="text-[10px] font-semibold border px-2 py-0.5 rounded-full ${badgeColor}">
                        ${sector.damageCount} dañados
                    </span>
                </div>
            `;
            
            sectorElement.addEventListener("click", () => {
                focusOnSector(sector);
            });
            
            listContainer.appendChild(sectorElement);
        });
    }

    generateScenario();
    populateSectorList();
</script>

<script>
    const canvas = document.getElementById("map-canvas");
    const ctx = canvas.getContext("2d");
    const container = document.getElementById("canvas-container");

    // Parámetros de transformación del mapa (Pan y Zoom)
    let zoom = 1.0;
    let offsetX = 50;
    let offsetY = 20;
    let isDragging = false;
    let startX, startY;

    // Filtros de estado actuales
    let currentSeverityFilter = "all";
    let baseLayerMode = "contour"; // 'contour' o 'sat'

    // Estructura seleccionada actualmente
    let hoveredBuilding = null;
    let selectedBuilding = null;

    // Ajustar dimensiones del canvas
    function resizeCanvas() {
        const rect = container.getBoundingClientRect();
        canvas.width = rect.width;
        canvas.height = rect.height;
        drawMap();
    }
    
    window.addEventListener("resize", resizeCanvas);
    setTimeout(resizeCanvas, 100);

    // Contornos topográficos artificiales para simular terreno 3D/curvas de nivel
    const CONTOURS = [];
    for (let i = 0; i < 12; i++) {
        // Generar curvas sinuosas que van de arriba (montaña) a abajo (playa)
        const points = [];
        const elevationY = 100 + i * 45;
        for (let x = -100; x <= 1100; x += 100) {
            // Añadir ruido ondulatorio para simular relieve natural
            const waveY = elevationY - 40 * Math.sin(x / 200) + 15 * Math.sin(x / 50);
            points.push({ x, y: waveY });
        }
        CONTOURS.push(points);
    }

    // Dibujar el mapa completo
    function drawMap() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        
        ctx.save();
        // Aplicar transformación de Pan & Zoom
        ctx.translate(offsetX, offsetY);
        ctx.scale(zoom, zoom);

        // 1. Dibujar Capa Base de Terreno
        if (baseLayerMode === "contour") {
            // Fondo oscuro para topografía tipo ArcGIS
            ctx.fillStyle = "#0c1324";
            ctx.fillRect(-1000, -1000, 3000, 3000);

            // Dibujar cuadrícula técnica de fondo
            ctx.strokeStyle = "rgba(51, 65, 85, 0.15)";
            ctx.lineWidth = 1;
            const gridSize = 50;
            for (let x = -500; x < 1500; x += gridSize) {
                ctx.beginPath(); ctx.moveTo(x, -500); ctx.lineTo(x, 1500); ctx.stroke();
            }
            for (let y = -500; y < 1500; y += gridSize) {
                ctx.beginPath(); ctx.moveTo(-500, y); ctx.lineTo(1500, y); ctx.stroke();
            }

            // Dibujar laderas montañosas de La Guaira (Curvas de nivel realistas)
            ctx.lineWidth = 1;
            CONTOURS.forEach((contourPoints, idx) => {
                ctx.strokeStyle = idx % 3 === 0 ? "rgba(34, 211, 238, 0.25)" : "rgba(34, 211, 238, 0.1)";
                ctx.beginPath();
                ctx.moveTo(contourPoints[0].x, contourPoints[0].y);
                for (let i = 1; i < contourPoints.length; i++) {
                    ctx.lineTo(contourPoints[i].x, contourPoints[i].y);
                }
                ctx.stroke();

                // Añadir etiqueta de elevación en alguna parte del trazado
                if (idx % 2 === 0) {
                    ctx.fillStyle = "rgba(34, 211, 238, 0.3)";
                    ctx.font = "8px 'JetBrains Mono'";
                    ctx.fillText(`${(12 - idx) * 20}m`, contourPoints[4].x, contourPoints[4].y - 3);
                }
            });

            // Dibujar zona del Mar Caribe en azul degradado
            const oceanGrad = ctx.createLinearGradient(500, 520, 500, 800);
            oceanGrad.addColorStop(0, "#083344");
            oceanGrad.addColorStop(1, "#02203c");
            ctx.fillStyle = oceanGrad;
            
            ctx.beginPath();
            ctx.moveTo(-200, 530);
            // Sinuosa costa artificial
            for (let x = -200; x <= 1200; x += 100) {
                ctx.lineTo(x, 530 + 15 * Math.sin(x / 120));
            }
            ctx.lineTo(1200, 1000);
            ctx.lineTo(-200, 1000);
            ctx.closePath();
            ctx.fill();

            // Línea de la playa destacada
            ctx.strokeStyle = "#22d3ee";
            ctx.lineWidth = 2;
            ctx.beginPath();
            ctx.moveTo(-200, 530);
            for (let x = -200; x <= 1200; x += 100) {
                ctx.lineTo(x, 530 + 15 * Math.sin(x / 120));
            }
            ctx.stroke();

        } else {
            // Fondo simulando fotografía satelital real de la costa de Vargas / La Guaira
            ctx.fillStyle = "#101b13"; // Verde bosque oscuro para la ladera
            ctx.fillRect(-1000, -1000, 3000, 3000);

            // Texturas montañosas y barrancos
            const mountainGrad = ctx.createLinearGradient(0, 0, 1000, 1000);
            mountainGrad.addColorStop(0, "#0e1e12");
            mountainGrad.addColorStop(0.5, "#162e1a");
            mountainGrad.addColorStop(1, "#1c2b16");
            ctx.fillStyle = mountainGrad;
            ctx.fillRect(-200, -200, 1400, 800);

            // Mar Profundo en Satelital
            ctx.fillStyle = "#01121d";
            ctx.beginPath();
            ctx.moveTo(-200, 530);
            for (let x = -200; x <= 1200; x += 100) {
                ctx.lineTo(x, 530 + 15 * Math.sin(x / 120));
            }
            ctx.lineTo(1200, 1000);
            ctx.lineTo(-200, 1000);
            ctx.closePath();
            ctx.fill();
            
            // Rompeolas y espuma de costa en blanco/cyan
            ctx.strokeStyle = "rgba(255, 255, 255, 0.4)";
            ctx.lineWidth = 3;
            ctx.setLineDash([5, 15]);
            ctx.beginPath();
            ctx.moveTo(-200, 532);
            for (let x = -200; x <= 1200; x += 100) {
                ctx.lineTo(x, 532 + 15 * Math.sin(x / 120));
            }
            ctx.stroke();
            ctx.setLineDash([]); // Restaurar
        }

        // 2. Dibujar Infraestructura Crítica: Aeropuerto de Maiquetía
        drawAirportRunways();

        // 3. Dibujar las Zonas de Conflictos (Límites circulares de riesgo sectorial)
        SECTOR_DATA.forEach(sector => {
            ctx.strokeStyle = sector.severity === "severo" ? "rgba(239, 68, 68, 0.15)" : "rgba(249, 115, 22, 0.15)";
            ctx.fillStyle = sector.severity === "severo" ? "rgba(239, 68, 68, 0.02)" : "rgba(249, 115, 22, 0.01)";
            ctx.lineWidth = 2;
            ctx.beginPath();
            ctx.arc(sector.x, sector.y, sector.radius, 0, Math.PI * 2);
            ctx.fill();
            ctx.stroke();

            // Etiqueta del Sector
            ctx.fillStyle = "rgba(255, 255, 255, 0.7)";
            ctx.font = "bold 9px 'Inter'";
            ctx.fillText(sector.name.toUpperCase(), sector.x - 30, sector.y);
        });

        // 4. Dibujar los Edificios Dañados (Micro-puntos de impacto estilo ArcGIS)
        GENERATED_BUILDINGS.forEach(building => {
            // Filtrar según el selector lateral de gravedad
            if (currentSeverityFilter !== "all" && building.severity !== currentSeverityFilter) {
                return;
            }

            // Colores correspondientes a la gravedad de mampostería dañada
            let color = "rgba(239, 68, 68, 0.85)"; // Severo (Rojo)
            if (building.severity === "moderate") color = "rgba(249, 115, 22, 0.85)"; // Moderado (Naranja)
            if (building.severity === "light") color = "rgba(234, 179, 8, 0.85)"; // Leve (Amarillo)

            // Si está siendo hovered, hacerlo más grande y brillante
            const isHovered = hoveredBuilding && hoveredBuilding.id === building.id;
            const isSelected = selectedBuilding && selectedBuilding.id === building.id;
            
            if (isHovered || isSelected) {
                ctx.shadowColor = "#ef4444";
                ctx.shadowBlur = 10;
                ctx.fillStyle = "#ffffff";
                ctx.beginPath();
                ctx.arc(building.x, building.y, 6, 0, Math.PI * 2);
                ctx.fill();
                ctx.shadowBlur = 0; // reset
                
                // Anillo exterior brillante
                ctx.strokeStyle = "#fca5a5";
                ctx.lineWidth = 1.5;
                ctx.stroke();
            } else {
                ctx.fillStyle = color;
                ctx.beginPath();
                // Dibujar micro cuadrados para simular edificios satelitales ortogonales
                ctx.fillRect(building.x - 2, building.y - 2, 4, 4);
            }
        });

        ctx.restore();
    }

    // Dibujar pistas del Aeropuerto de Maiquetía
    function drawAirportRunways() {
        ctx.save();
        // Plataforma asfalto principal (Runway 10-28)
        ctx.fillStyle = "rgba(71, 85, 105, 0.8)";
        ctx.strokeStyle = "rgba(148, 163, 184, 0.3)";
        ctx.lineWidth = 1;
        
        // Pista 1 (Frente al mar)
        ctx.fillRect(150, 410, 480, 18);
        ctx.strokeRect(150, 410, 480, 18);

        // Pista 2 (Interna)
        ctx.fillRect(180, 440, 400, 14);
        ctx.strokeRect(180, 440, 400, 14);

        // Marcaciones de pista didácticas (Línea blanca punteada)
        ctx.strokeStyle = "rgba(255, 255, 255, 0.7)";
        ctx.lineWidth = 1;
        ctx.setLineDash([8, 12]);
        ctx.beginPath();
        ctx.moveTo(160, 419); ctx.lineTo(620, 419);
        ctx.moveTo(190, 447); ctx.lineTo(570, 447);
        ctx.stroke();
        ctx.setLineDash([]); // Reset

        // Grietas sísmicas simuladas en la pista (Líneas negras irregulares)
        ctx.strokeStyle = "rgba(0, 0, 0, 0.85)";
        ctx.lineWidth = 2;
        ctx.beginPath();
        ctx.moveTo(280, 410); ctx.lineTo(290, 418); ctx.lineTo(285, 428);
        ctx.moveTo(410, 411); ctx.lineTo(412, 419); ctx.lineTo(418, 422);
        ctx.moveTo(490, 439); ctx.lineTo(495, 448); ctx.lineTo(491, 454);
        ctx.stroke();

        ctx.restore();
    }

    // Control de Arrastre (Panning) del Canvas
    canvas.addEventListener("mousedown", (e) => {
        isDragging = true;
        startX = e.clientX - offsetX;
        startY = e.clientY - offsetY;
    });

    window.addEventListener("mouseup", () => {
        isDragging = false;
    });

    canvas.addEventListener("mousemove", (e) => {
        const rect = canvas.getBoundingClientRect();
        const mouseX = e.clientX - rect.left;
        const mouseY = e.clientY - rect.top;

        if (isDragging) {
            offsetX = e.clientX - startX;
            offsetY = e.clientY - startY;
            drawMap();
        } else {
            // Detección de colisión para hover en edificios
            detectHover(mouseX, mouseY);
        }

        // Actualizar coordenadas en el HUD flotante
        updateCoordHUD(mouseX, mouseY);
    });

    // Detectar si el cursor está sobre un punto crítico de daño
    function detectHover(mx, my) {
        // Convertir coordenadas de pantalla a coordenadas del modelo transformado
        const worldX = (mx - offsetX) / zoom;
        const worldY = (my - offsetY) / zoom;

        let found = null;
        for (let i = GENERATED_BUILDINGS.length - 1; i >= 0; i--) {
            const b = GENERATED_BUILDINGS[i];
            if (currentSeverityFilter !== "all" && b.severity !== currentSeverityFilter) continue;
            
            const dist = Math.hypot(b.x - worldX, b.y - worldY);
            if (dist < 8 / zoom) { // tolerancia de click/hover de 8 pixeles
                found = b;
                break;
            }
        }

        if (found !== hoveredBuilding) {
            hoveredBuilding = found;
            canvas.style.cursor = found ? "pointer" : (isDragging ? "grabbing" : "grab");
            drawMap();
        }
    }

    // Click en el Canvas para seleccionar y abrir detalles
    canvas.addEventListener("click", (e) => {
        if (hoveredBuilding) {
            selectedBuilding = hoveredBuilding;
            showDetailsModal(selectedBuilding);
        } else {
            selectedBuilding = null;
            document.getElementById("detail-modal").classList.add("hidden");
        }
        drawMap();
    });

    // Mostrar el Modal de Información de Estructuras
    function showDetailsModal(b) {
        document.getElementById("modal-title").textContent = `Estructura ID: ${b.id}`;
        document.getElementById("modal-sector").textContent = b.sectorName;
        document.getElementById("modal-elevation").textContent = b.elev;
        document.getElementById("modal-usage").textContent = b.usage;
        document.getElementById("modal-desc").textContent = b.desc;

        let sevText = "Moderado";
        let sevClass = "text-orange-400";
        if (b.severity === "severe") {
            sevText = "Severo / Colapso Crítico";
            sevClass = "text-red-500 font-bold";
        } else if (b.severity === "light") {
            sevText = "Leve";
            sevClass = "text-yellow-400";
        }
        
        const sevElement = document.getElementById("modal-severity");
        sevElement.textContent = sevText;
        sevElement.className = sevClass;

        document.getElementById("detail-modal").classList.remove("hidden");
    }

    // Cerrar modal
    document.getElementById("modal-close").addEventListener("click", () => {
        document.getElementById("detail-modal").classList.add("hidden");
        selectedBuilding = null;
        drawMap();
    });

    // Actualizar indicador de Latitud/Longitud didáctico
    function updateCoordHUD(mx, my) {
        const worldX = (mx - offsetX) / zoom;
        const worldY = (my - offsetY) / zoom;

        // Mapeo simple de elevación basado en coordenada Y (entre más arriba, más alto)
        let simulatedElev = Math.max(0, Math.floor((530 - worldY) * 0.4));
        if (worldY > 530) simulatedElev = 0; // Nivel del Mar

        document.getElementById("live-elev").textContent = `${simulatedElev} m`;

        // Detección de zona sobre la que flota
        let zoneText = "Fuera del área sismogénica";
        SECTOR_DATA.forEach(sector => {
            const dist = Math.hypot(sector.x - worldX, sector.y - worldY);
            if (dist < sector.radius) {
                zoneText = `Sector: ${sector.name}`;
            }
        });
        document.getElementById("hovered-zone").textContent = zoneText;
    }

    // Zoom táctil o con scroll de mouse
    canvas.addEventListener("wheel", (e) => {
        e.preventDefault();
        const zoomFactor = 1.1;
        if (e.deltaY < 0) {
            zoom = Math.min(zoom * zoomFactor, 8.0);
        } else {
            zoom = Math.max(zoom / zoomFactor, 0.4);
        }
        drawMap();
    });

    // Enfoque automático de cámara al hacer click en sectores de la barra lateral
    function focusOnSector(sector) {
        const rect = canvas.getBoundingClientRect();
        zoom = 1.5;
        // Centrar coordenada del sector en el centro del viewport del canvas
        offsetX = rect.width / 2 - sector.x * zoom;
        offsetY = rect.height / 2 - sector.y * zoom;
        drawMap();

        // Resaltar sector seleccionando simbólicamente el primer edificio de este
        const sectorBuildings = GENERATED_BUILDINGS.filter(b => b.sectorName === sector.name);
        if (sectorBuildings.length > 0) {
            selectedBuilding = sectorBuildings[0];
            showDetailsModal(selectedBuilding);
        }
    }

    // Botones de Zoom flotantes
    document.getElementById("map-zoom-in").addEventListener("click", () => {
        zoom = Math.min(zoom * 1.3, 8.0);
        drawMap();
    });
    document.getElementById("map-zoom-out").addEventListener("click", () => {
        zoom = Math.max(zoom / 1.3, 0.4);
        drawMap();
    });
    document.getElementById("map-reset").addEventListener("click", () => {
        zoom = 1.0;
        offsetX = 50;
        offsetY = 20;
        selectedBuilding = null;
        document.getElementById("detail-modal").classList.add("hidden");
        drawMap();
    });

    document.getElementById("severity-filter").addEventListener("change", (e) => {
        currentSeverityFilter = e.target.value;
        
        // Actualizar estadísticas del Header basándose en el filtro
        const statLabel = document.getElementById("stat-total-damaged");
        if (currentSeverityFilter === "all") statLabel.textContent = "3,421";
        else if (currentSeverityFilter === "severe") statLabel.textContent = "1,104";
        else if (currentSeverityFilter === "moderate") statLabel.textContent = "1,512";
        else if (currentSeverityFilter === "light") statLabel.textContent = "805";

        drawMap();
    });

        baseLayerMode = "sat";

        btnSat.classList.remove("text-slate-400", "border-slate-800");
        
        btnContour.classList.remove("selected", "text-white", "border-slate-700/60");
        btnContour.classList.add("text-slate-400", "border-slate-800");
        drawMap();
    });
